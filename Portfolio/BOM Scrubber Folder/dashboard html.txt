<script>
  

function showLoader(message){

  document.getElementById(
    "loader"
  ).style.display = "block";

  document.getElementById(
    "loader"
  ).innerHTML = message;

  document.getElementById(
    "statusCard"
  ).innerHTML = "RUNNING";

  document.getElementById(
  "statusCard"
).style.color = "orange";

}

function validateBom(){

  showLoader(
    "⏳ Validating BOM..."
  );

  google.script.run
  .withSuccessHandler(function(){

    hideLoader();

  })
  .validateImportedBom();

}

function hideLoader(){

  document.getElementById(
    "loader"
  ).style.display = "none";

  document.getElementById(
    "statusCard"
  ).innerHTML = "COMPLETED";

  document.getElementById(
    "statusCard"
  ).style.color = "green";

}

google.script.run
.withSuccessHandler(loadData)
.getDashboardData();

function loadData(data){

document.getElementById("api")
.innerHTML=data.api;

document.getElementById("parts")
.innerHTML=data.totalParts;

document.getElementById("descMatch")
.innerHTML =
data.descMatch;

document.getElementById("descMismatch")
.innerHTML =
data.descMismatch;

document.getElementById("packageMatch")
.innerHTML =
data.packageMatch;

document.getElementById("packageMismatch")
.innerHTML =
data.packageMismatch;

document.getElementById("datasheets")
.innerHTML=data.datasheets;

document.getElementById("lastRun")
.innerHTML = data.lastRun;

document.getElementById(
"summaryText"
).innerHTML =

"Processed : " +
data.totalParts +

" | Desc Mismatch : " +
data.descMismatch +

" | Package Mismatch : " +
data.packageMismatch;

document.getElementById(
"statusCard"
).innerHTML = "READY";

}

google.script.run
  .withSuccessHandler(loadData)
  .getDashboardData();

  

function runScrub(){

  showLoader(
    "⏳ BOM Scrubbing In Progress..."
  );

  google.script.run

  .withSuccessHandler(function(){

    hideLoader();

    alert(
      "✅ BOM Scrubbing Completed"
    );

    // google.script.host.close();

    // google.script.run.showDashboard();

  })

  .fullScrub();

}

function downloadScrubReport(){

  google.script.run
    .withSuccessHandler(function(url){

      window.open(
        url,
        "_blank"
      );

    })
    .exportScrubReport();

}


function openScrubOutput(){

  google.script.run.openScrubOutput();

}

function showError(){

  document.getElementById(
    "statusCard"
  ).innerHTML = "FAILED";

  document.getElementById(
    "statusCard"
  ).style.color = "red";

}
