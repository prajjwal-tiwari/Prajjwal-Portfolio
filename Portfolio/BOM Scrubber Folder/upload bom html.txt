function uploadBomFile(){

    document.getElementById(

    ).files[0];

      "status"
    ).innerHTML =
    "❌ Please Select BOM File";

    return;

  }

  document.getElementById(
    "status"
  ).innerHTML =
  "✅ File Selected : " +
  file.name;
